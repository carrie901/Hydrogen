190  /// <summary>
191  /// Process / Update our AudioStackItem
192  /// </summary>
193  public void Process ()
194  {
195  if (Source == null || Clip == null)
196  return;
197 
198  if (TargetVolume > MaxVolume)
199  TargetVolume = MaxVolume;
200 
201  if (Source.volume != TargetVolume) {
202  if (Fade) {
203  if (Source.volume > TargetVolume) {
204  Source.volume =
205  Mathf.Lerp (Source.volume, TargetVolume, FadeOutSpeed * Time.deltaTime);
206  } else {
207  Source.volume =
208  Mathf.Lerp (Source.volume, TargetVolume, FadeInSpeed * Time.deltaTime);
209  }
210  } else {
211  Source.volume = TargetVolume;
212  }
213  }
214 
215  // Automatically remove finished processes, but only if they are not marked persistant.
216  // Not checking for Loop as if someone sets it's TargetVolume to 0 its meant to go away.
217  if (!Persistant && ((Fade && RemoveAfterFadeOut && Source.volume < 0.0001f) || Source.time == Clip.length)) {
218 
219  Stack.Remove (this);
220  }
221  }
