110  /// <summary>
111  /// Enqueues an operation.
112  /// </summary>
113  /// <param name="action">The operation to perform.</param>
114  /// <param name="state">An opaque object, allowing the caller to identify operations.</param>
115  /// <returns>The current instance (for fluent API).</returns>
116  public Forker Fork (ThreadStart action, object state)
117  {
118  if (action == null)
119  throw new ArgumentNullException ("action");
120  Interlocked.Increment (ref _runningCount);
121  ThreadPool.QueueUserWorkItem (delegate {
122  Exception exception = null;
123  try {
124  action ();
125  } catch (Exception ex) {
126  exception = ex;
127  }
128  OnItemComplete (state, exception);
129  });
130  return this;
131  }

141  /// <summary>
142  /// Waits (with timeout) for all operations to complete.
143  /// </summary>
144  /// <returns>Whether all operations had completed before the timeout.</returns>
145  public bool Join (int millisecondsTimeout)
146  {
147  lock (_joinLock) {
148  if (CountRunning () == 0)
149  return true;
150  Thread.SpinWait (1); // try our luck...
151  return (CountRunning () == 0) ||
152  Monitor.Wait (_joinLock, millisecondsTimeout);
153  }
154  }

180  /// <summary>
181  /// Raised when an item is completed.
182  /// </summary>
183  /// <param name="state">Item State.</param>
184  /// <param name="exception">Exception.</param>
185  void OnItemComplete (object state, Exception exception)
186  {
187  EventHandler<ParallelEventArgs> itemHandler = _itemComplete; // don't need to lock
188  if (itemHandler != null)
189  itemHandler (this, new ParallelEventArgs (state, exception));
190  if (Interlocked.Decrement (ref _runningCount) == 0) {
191  EventHandler allHandler = _allComplete; // don't need to lock
192  if (allHandler != null)
193  allHandler (this, EventArgs.Empty);
194  lock (_joinLock) {
195  Monitor.PulseAll (_joinLock);
196  }
197  }
198  }
