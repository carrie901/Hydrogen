97  case TypeCode.Int16:

99  case TypeCode.Int32:
100  return (zeroIsPositive ? (int)value >= 0 : (int)value > 0);
101  case TypeCode.Int64:
102  return (zeroIsPositive ? (long)value >= 0 : (long)value > 0);
103  case TypeCode.Single:
104  return (zeroIsPositive ? (float)value >= 0 : (float)value > 0);
105  case TypeCode.Double:
106  return (zeroIsPositive ? (double)value >= 0 : (double)value > 0);
107  case TypeCode.Decimal:
108  return (zeroIsPositive ? (decimal)value >= 0 : (decimal)value > 0);
109  case TypeCode.Byte:
110  return (zeroIsPositive ? true : (byte)value > 0);
111  case TypeCode.UInt16:
112  return (zeroIsPositive ? true : (ushort)value > 0);

115  case TypeCode.UInt64:
116  return (zeroIsPositive ? true : (ulong)value > 0);
