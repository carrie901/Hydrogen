38  /// <summary>
39  /// Autocrop the texture removes the borders from the texture. It searches the active layer for the
40  /// largest possible border area that's color matches borderColor, and then crops this area from the
41  /// image, as if you had used the Crop tool.
42  /// </summary>
43  /// <returns>The cropped image.</returns>
44  /// <param name="texture">The source texture.</param>
45  /// <param name="borderColor">The border color used in the crop calculations.</param>
46  public static Texture2D AutoCrop (this Texture2D texture, Color borderColor)
47  {
48  var image = texture.GetPixels ();
49  var width = texture.width;
50  var height = texture.height;
51 
52  var left = -1;
53  var right = -1;
54  var top = -1;
55  var bottom = -1;
56  var x = 0;
57  int y;
58 
59  while (left == -1 && x < width) {
60  for (y = 0; y < height; y++) {
61  if (!image [y * width + x].IsColorApproximatelySame (borderColor))
62  left = x;
63  }
64  x++;
65  }
66 
67  x = width - 1;
68 
69  while (right == -1 && x >= 0) {
70  for (y = 0; y < height; y++) {
71  if (!image [y * width + x].IsColorApproximatelySame (borderColor))
72  right = x;
73  }
74  x--;
75  }
76 
77  y = 0;
78 
79  while (top == -1 && y < height) {
80  for (x = 0; x < width; x++) {
81  if (!image [y * width + x].IsColorApproximatelySame (borderColor))
82  top = y;
83  }
84  y++;
85  }
86 
87  y = height - 1;
88 
89  while (bottom == -1 && y >= 0) {
90  for (x = 0; x < width; x++) {
91  if (!image [y * width + x].IsColorApproximatelySame (borderColor))
92  bottom = y;
93  }
94  y--;
95  }
96 
97  return Crop (texture, left, top, right - left, bottom - top);
98  }

100  /// <summary>
101  /// The pixels (RGB) of both the destinationTexture and the sourceTexture are combined bitwise according to
102  /// the simplified overwrite raster operation, and the result is then written to the destinationTexture.
103  /// </summary>
104  /// <param name="destinationTexture">The destination texture.</param>
105  /// <param name="sourceTexture">The source texture.</param>
106  /// <param name="offsetX">Offset X.</param>
107  /// <param name="offsetY">Offset Y.</param>
108  public static void Blit (Texture2D destinationTexture, Texture2D sourceTexture, int offsetX, int offsetY)
109  {
110  var srcWidth = sourceTexture.width;
111  // int srcHeight = src.height;
112  var dstWidth = destinationTexture.width;
113  // int dstHeight = dst.height;
114  Color[] dstp = destinationTexture.GetPixels ();
115  Color[] srcp = sourceTexture.GetPixels ();
116 
117  for (int y = 0; y < sourceTexture.height; y++) {
118  for (int x = 0; x < sourceTexture.width; x++) {
119  dstp [(y + offsetY) * dstWidth + x + offsetX] = srcp [y * srcWidth + x];
120  }
121  }
122 
123  destinationTexture.SetPixels (dstp);
124  }

126  /// <summary>
127  /// The pixels (RGBA) of both the destinationTexture and the sourceTexture are combined bitwise according to
128  /// the simplified overwrite raster operation, and the result is then written to the destinationTexture.
129  /// </summary>
130  /// <param name="destinationTexture">The destination texture.</param>
131  /// <param name="sourceTexture">The source texture.</param>
132  /// <param name="offsetX">Offset X.</param>
133  /// <param name="offsetY">Offset Y.</param>
134  public static void BlitAlpha (Texture2D destinationTexture, Texture2D sourceTexture, int offsetX, int offsetY)
135  {
136  try {
137  var srcWidth = sourceTexture.width;
138  var srcHeight = sourceTexture.height;
139  var dstWidth = destinationTexture.width;
140  var dstps = destinationTexture.GetPixels ();
141  var srcps = sourceTexture.GetPixels ();
142  Color srcp;
143 
144  for (int y = 0; y < srcHeight; y++) {
145  for (int x = 0; x < srcWidth; x++) {
146  int dstOffset = (y + offsetY) * dstWidth + x + offsetX;
147  int srcOffset = y * srcWidth + x;
148  srcp = srcps [srcOffset];
149  if (srcp.a > 0.001f) {
150  dstps [dstOffset] = Color.Lerp (dstps [dstOffset], srcp, srcp.a);
151  }
152  }
153  }
154 
155  destinationTexture.SetPixels (dstps);
156  } catch (System.Exception e) {
157  Debug.Log ("[H] There was a problem Blitting the texture. " + e);
158  }
159  }

230  /// <summary>
231  /// Resize the texture to a specific width and height using bilinear filtering.
232  /// </summary>
233  /// <param name="sourceTexture">The source texture.</param>
234  /// <param name="newWidth">The new resized width.</param>
235  /// <param name="newHeight">The new resized height.</param>
236  public static Texture2D Resize (this Texture2D sourceTexture, int newWidth, int newHeight)
237  {
238  var resizedTexture = new Texture2D (newWidth, newHeight);
239  var resizeFactor = new Vector2 ((float)sourceTexture.width / (float)newWidth, (float)sourceTexture.height / (float)newHeight);
240 
241  Vector2 fraction;
242  Vector2 oneMinusFraction;
243  var one = new Vector2 (1f, 1f);
244 
245  Color c1;
246  Color c2;
247  Color c3;
248  Color c4;
249 
250  Color[] image = sourceTexture.GetPixels ();
251  Color[] result = resizedTexture.GetPixels ();
252 
253  for (var y = 0; y < newHeight; y++) {
254  for (var x = 0; x < newWidth; x++) {
255  var floorX = (int)Mathf.Floor (x * resizeFactor.x);
256  var floorY = (int)Mathf.Floor (y * resizeFactor.y);
257 
258  var ceilingX = floorX + 1;
259 
260  if (ceilingX >= sourceTexture.width)
261  ceilingX = floorX;
262 
263  var ceilingY = floorY + 1;
264  if (ceilingY >= sourceTexture.height)
265  ceilingY = floorY;
266 
267  fraction = new Vector2 (x * resizeFactor.x - floorX, y * resizeFactor.y - floorY);
268  oneMinusFraction = one - fraction;
269 
270  c1 = image [floorY * sourceTexture.width + floorX];
271  c2 = image [floorY * sourceTexture.width + ceilingX];
272  c3 = image [ceilingY * sourceTexture.width + floorX];
273  c4 = image [ceilingY * sourceTexture.width + ceilingX];
274 
275  // Blue
276  float b1 = (oneMinusFraction.x * c1.b + fraction.x * c2.b);
277  float b2 = (oneMinusFraction.x * c3.b + fraction.x * c4.b);
278 
279  float blue = (oneMinusFraction.y * b1 + fraction.y * b2);
280 
281  // Green
282  b1 = (oneMinusFraction.x * c1.g + fraction.x * c2.g);
283  b2 = (oneMinusFraction.x * c3.g + fraction.x * c4.g);
284 
285  float green = (oneMinusFraction.y * b1 + fraction.y * b2);
286 
287  // Red
288  b1 = (oneMinusFraction.x * c1.r + fraction.x * c2.r);
289  b2 = (oneMinusFraction.x * c3.r + fraction.x * c4.r);
290 
291  float red = (oneMinusFraction.y * b1 + fraction.y * b2);
292 
293  // Alpha
294  b1 = (oneMinusFraction.x * c1.a + fraction.x * c2.a);
295  b2 = (oneMinusFraction.x * c3.a + fraction.x * c4.a);
296 
297  float alpha = (oneMinusFraction.y * b1 + fraction.y * b2);
298 
299  result [y * resizedTexture.width + x] = new Color (red, green, blue, alpha);
300  }
301  }
302 
303  resizedTexture.SetPixels (result);
304 
305  return resizedTexture;
306  }

360  /// <summary>

362  /// </summary>

364  /// <param name="texture">The texture to tile.</param>
365  /// <param name="horizontalTiles">Number of horizontal tiles.</param>
366  /// <param name="verticalTiles">Number of vertical tiles.</param>
367  public static Texture2D Tile (this Texture2D texture, int horizontalTiles, int verticalTiles)
368  {
369  // Sanity Check
370  if (horizontalTiles < 1)
371  horizontalTiles = 1;
372  if (verticalTiles < 1)
373  verticalTiles = 1;
374 
375  var tiledTexture = new Texture2D (texture.width * horizontalTiles, texture.height * verticalTiles);
376  var texturePixels = texture.GetPixels ();
377  for (var i = 0; i < horizontalTiles; i++) {
378  for (var j = 0; j < verticalTiles; j++)
379  tiledTexture.SetPixels (i * texture.width, j * texture.height, texture.width, texture.height, texturePixels);
380  }
381 
382  return tiledTexture;
