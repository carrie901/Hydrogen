41  /// <returns><c>true</c>, if at destination was agented, <c>false</c> otherwise.</returns>
42  /// <param name="agent">The target NavMeshAgent.</param>
43  public static bool AgentAtDestination (this NavMeshAgent agent)
44  {
45  float dist = agent.remainingDistance;
46  if (!float.IsPositiveInfinity (dist) && agent.pathStatus == NavMeshPathStatus.PathComplete &&
47  agent.remainingDistance == 0f) {
48  return true;
49  }
50  return false;
51  }
