74  _hash = (Time.time + URI + formStringData.GetHashCode () + Random.Range (0, 100)).GetHashCode ();

142  /// <summary>
143  /// IEnumeratable HTTP POST Form to URI.
144  /// </summary>
145  /// <returns>IEnumeratable Function</returns>
146  /// <param name="URI">The Target URI.</param>
147  /// <param name="formStringData">A Dictionary<string,string> of Form Data</param>
148  /// <param name="formBinaryData">A custom binary dataset. Useful for uploading pictures.</param>
149  /// <param name="cookie">Any previous cookie data to be used for authentication.</param>
150  /// <param name="callback">A callback function (int hash, Hashtable headers, string payload).</param>
151  IEnumerator FormReturnedText (string URI, Dictionary<string, string> formStringData, WebPool.FormBinaryData[] formBinaryData, string cookie, System.Action<int, Hashtable, string> callback)
152  {
153  // Assign Busy Flag
154  _busy = true;
155 
156  var newForm = new WWWForm ();
157 
158  // Add string data
159  foreach (string s in formStringData.Keys) {
160  newForm.AddField (s, formStringData [s]);
161  }
162 
163  // Add binary data
164  foreach (WebPool.FormBinaryData b in formBinaryData) {
165  newForm.AddBinaryData (b.FieldName, b.Data, b.FileName, b.MimeType);
166  }
167 
168  var headers = newForm.headers;
169 
170  if (cookie != null)
171  headers.Add ("Cookie", cookie);
172 
173  var newCall = new WWW (URI, newForm.data, headers);
174 
175  yield return newCall;
176 
177  while (!newCall.isDone)
178  yield return new WaitForSeconds (0.01f);
179 
180  // Callback!
181  if (callback != null) {
182  if (newCall.responseHeaders ["STATUS"].Contains (" 200 "))
183  callback (_hash, new Hashtable (newCall.responseHeaders), newCall.text);
184  else
185  callback (_hash, new Hashtable (newCall.responseHeaders), "");
186  }
187 
188  _busy = false;
189 
190  ParentPool.Despawn (gameObject);
191  }

193  /// <summary>
194  /// IEnumeratable HTTP GET Request to URI.
195  /// </summary>
196  /// <returns>Enumeratable Function</returns>
197  /// <param name="URI">The Target URI.</param>
198  /// <param name="cookie">Any previous cookie data to be used for authentication.</param>
199  /// <param name="callback">A callback function (int hash, Hashtable headers, string payload).</param>
200  IEnumerator GetReturnedText (string URI, string cookie, System.Action<int, Hashtable, string> callback)
201  {
202  // Assign Busy Flag
203  _busy = true;
204 
205  // Process Headers
206  var headers = new Hashtable ();
207  if (cookie != null)
208  headers.Add ("Cookie", cookie);
209 
210  // Make the call
211  var newCall = new WWW (URI, null, headers);
212 
213  yield return newCall;
214 
215  while (!newCall.isDone)
216  yield return new WaitForSeconds (0.01f);
217 
218  // Callback! (Avoid Unity Bitching)
219  if (callback != null) {
220  if (newCall.responseHeaders ["STATUS"].Contains (" 200 "))
221  callback (_hash, new Hashtable (newCall.responseHeaders), newCall.text);
222  else
223  callback (_hash, new Hashtable (newCall.responseHeaders), "");
224  }
225 
226  _busy = false;
227

231  /// <summary>
232  /// IEnumeratable HTTP POST to URI.
233  /// </summary>
234  /// <returns>Enumeratable Function</returns>
235  /// <param name="URI">The Target URI.</param>
236  /// <param name="contentType">The Content-Type Header</param>
237  /// <param name="payload">The data to be posted.</param>
238  /// <param name="cookie">Any previous cookie data to be used for authentication.</param>
239  /// <param name="callback">A callback function (int hash, Hashtable headers, string payload).</param>
240  IEnumerator PostReturnedText (string URI, string contentType, string payload, string cookie, System.Action<int, Hashtable, string> callback)
241  {
242  // Assign Busy Flag
243  _busy = true;
244 
245  // Message Data
246  byte[] postData = System.Text.Encoding.ASCII.GetBytes (payload.ToCharArray ());
247 
248  // Process Headers
249  var headers = new Hashtable ();
250 
251  headers.Add ("Content-Type", contentType);
252  headers.Add ("Content-Length", postData.Length);
253  if (cookie != null)
254  headers.Add ("Cookie", cookie);
255 
256  var newCall = new WWW (URI, postData, headers);
257 
258  yield return newCall;
259 
260  while (!newCall.isDone)
261  yield return new WaitForSeconds (0.01f);
262 
263  if (callback != null) {
264  if (newCall.responseHeaders ["STATUS"].Contains (" 200 "))
265  callback (_hash, new Hashtable (newCall.responseHeaders), newCall.text);
266  else
267  callback (_hash, new Hashtable (newCall.responseHeaders), "");
268  }
269 
270  _busy = false;
271 
272  ParentPool.Despawn (gameObject);
273  }
274  }
