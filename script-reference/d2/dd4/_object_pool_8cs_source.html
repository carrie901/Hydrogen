142  /// <summary>
143  /// Add a Prefab or GameObject to the Object Pool.
144  /// </summary>
145  /// <returns>
146  /// The index (Pool ID) of the newly created pool, or if a pool already exists, return its index.
147  /// </returns>
148  /// <remarks>
149  /// If you pass an existing GameObject from the scene, it will need to exist in the future to be able
150  /// to spawn any additional objects outside of what was preloaded upon its addition to the Object Pool.
151  /// It can be disabled, or inactive, it simply needs to exist if you want to add more later.
152  /// </remarks>
153  /// <param name="prefab">The Prefab or GameObject.</param>
154  /// <param name="preloadAmount">The number of objects that should be preloaded.</param>
155  /// <param name="spawnMore">Should more objects be added to the pool as needed?</param>
156  /// <param name="slowMessage">Should Unity's SendMessage be used?</param>
157  /// <param name="handleParticles">Should accomodations for Particles be made?</param>
158  /// <param name="trackSpawned">Should spawned objects be tracked?</param>
159  /// <param name="cullExtras">Should extra objects be removed from the Object Pool when not in use?</param>
160  /// <param name="cullInterval">How often should we check for extras to cull.</param>
161  public int Add (GameObject prefab, int preloadAmount, bool spawnMore, bool slowMessage,
162  bool handleParticles, bool trackSpawned, bool cullExtras, float cullInterval)
163  {
164  if (prefab == null) {
165  throw new MissingReferenceException ("You are passing a null gameObject reference to hObjectPool.Instance.Add()");
166  }
167 
168  var tempLookup = GetPoolID (prefab.name);
169  if (tempLookup == -1) {
170  var newPool = new ObjectPoolCollection (preloadAmount, spawnMore, slowMessage,
171  handleParticles, trackSpawned, cullExtras, cullInterval);
172  newPool.Initialize (prefab, transform, ObjectPools.Length);
173  Array.Add<ObjectPoolCollection> (ref ObjectPools, newPool, false);
174  if (_poolStringLookupTable == null)
175  _poolStringLookupTable = new Dictionary<string, int> ();
176  _poolStringLookupTable.Add (newPool.Prefab.name, ObjectPools.Length - 1);
177  return ObjectPools.Length - 1;
178  }
179  return tempLookup;
180  }

239  bool handleParticles, bool trackSpawned, bool cullExtras, float cullInterval)
240  {
241  var returnIDs = new int[prefabs.Length];
242 
243  for (int x = 0; x < prefabs.Length; x++) {
244  returnIDs [x] = Add (prefabs [x], preloadAmount, spawnMore, slowMessage, handleParticles,

308  /// <returns>The Pool ID, -1 if not found.</returns>
309  /// <param name="prefabName">Lookup Name</param>
310  public int GetPoolID (string prefabName)
311  {
312  if (string.IsNullOrEmpty (prefabName)) {
313  throw new MissingReferenceException (
314  "You are passing a null or empty prefabName to hObjectPool.Instance.GetPoolID()");
315  }
316  if (_poolStringLookupTable == null)
317  return -1;
318 
319  return _poolStringLookupTable.ContainsKey (prefabName) ? _poolStringLookupTable [prefabName] : -1;
320  }

322  /// <summary>
323  /// Remove the specified GameObject from it's Object Pool.
324  /// </summary>
325  /// <param name="gameObject">Target GameObject</param>
326  /// <param name="destroyImmediate">Should the object be destroied immediately.</param>
327  public bool Remove (GameObject gameObject, bool destroyImmediate)
328  {
329  var tempLookup = GetPoolID (gameObject.name);
330  if (tempLookup != -1) {
331  // Remove object from lookup table
332  _poolStringLookupTable.Remove (gameObject.name);
333 
334  // Remove from pooling system
335  return ObjectPools [tempLookup].RemoveFromPool (gameObject, destroyImmediate);
336  }
337  return false;
338  }

412  /// <summary>
413  /// Unity's Awake Event
414  /// </summary>
415  protected virtual void Awake ()
416  {
417  if (ObjectPools == null)
418  ObjectPools = new ObjectPoolCollection[0];
419  if (_poolStringLookupTable == null)
420  _poolStringLookupTable = new Dictionary<string, int> ();
421 
422  for (int x = 0; x < ObjectPools.Length; x++) {
423 
424  if (ObjectPools [x].Prefab == null) {
425  throw new MissingReferenceException ("You have not set the prefab in the hObjectPool Inspector for Object Pools->Element " + x);
426  }
427 
428  if (!ObjectPools [x].Initialized) {
429  ObjectPools [x].Initialize (ObjectPools [x].Prefab, transform, x);
430  _poolStringLookupTable.Add (ObjectPools [x].Prefab.name, x);
431  }
432  }
433  }
