36  /// <summary>
37  /// An internal pooling system used within the Hydrogen Framework.

56  public void Awake()
57  {
58  if (objectPools == null) objectPools = new Hydrogen.Core.ObjectPoolCollection[0];
59  if ( _poolStringLookupTable == null ) _poolStringLookupTable = new Dictionary<string, int>();
60 
61 
62  for (int x = 0; x < objectPools.Length; x++ )
63  {
64 
65  if ( objectPools[x].prefab == null )
66  {
67  throw new MissingReferenceException("You have not set the prefab in the hObjectPool Inspector for Object Pools->Element " + x.ToString());
68  }
69 
70  if ( !objectPools[x].Initialized )
71  {
72  objectPools[x].Initialize(objectPools[x].prefab, this.transform, x);
73  _poolStringLookupTable.Add(objectPools[x].prefab.name,x);
74  }
75  }
76  }

126  public int Add(GameObject gameObject, int preload, bool spawnMore, bool slowMessage, bool handleParticles, bool trackSpawned, bool cullExtras)
127  {
128  if ( gameObject == null ) {
129  throw new MissingReferenceException("You are passing a null gameObject reference to hObjectPool.Instance.Add()");
130  }
131 
132  int tempLookup = GetPoolID(gameObject.name);
133  if ( tempLookup == -1 )
134  {
135  // Create our new pool
136  Hydrogen.Core.ObjectPoolCollection newPool = new Hydrogen.Core.ObjectPoolCollection(preload, spawnMore, slowMessage, handleParticles, trackSpawned, cullExtras);
137 
138  // Initialize pool
139  newPool.Initialize(gameObject, this.transform, objectPools.Length);
140 
141  // Add a pool for the new object
142  Hydrogen.Array.Add<Hydrogen.Core.ObjectPoolCollection>(ref objectPools, newPool, false);
143 
144  // Add reference for lookup table
145  if ( _poolStringLookupTable == null ) _poolStringLookupTable = new Dictionary<string, int>();
146 
147  _poolStringLookupTable.Add (newPool.prefab.name, objectPools.Length - 1);
148 
149  return objectPools.Length - 1;
150  }
151  return tempLookup;
152  }

155  public bool Remove(GameObject gameObject, bool destroyImmediate)
156  {
157  int tempLookup = GetPoolID(gameObject.name);
158  if ( tempLookup != -1 )
159  {
160  // Remove object from lookup table
161  _poolStringLookupTable.Remove(gameObject.name);
162 
163  // Remove from pooling system
164  return objectPools[tempLookup].RemoveFromPool(gameObject, destroyImmediate);
165  }
166  return false;
167  }
