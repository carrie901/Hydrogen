170  /// <summary>
171  /// Initialization process for a WebPool.
172  /// </summary>
173  IEnumerator Initialize ()
174  {
175  if (!_initialized) {
176 
177  // Create our buddy object
178  var newWebObject = new GameObject ();
179  newWebObject.AddComponent (typeof(WebPoolWorker));
180  newWebObject.name = "Web Call";
181 
182  // Search out any existing ObjectPool
183  _poolReference = (ObjectPool)FindObjectOfType (typeof(ObjectPool));
184 
185  // If we don't have an existing reference in the scene for an ObjectPool, we need to make one.
186  if (_poolReference == null) {
187  // Create a new ObjectPool using our default singleton.
188  _poolReference = hObjectPool.Instance;
189 
190  // Wait for end of frame so that the new Object Pool can initialize.
191  yield return new WaitForEndOfFrame ();
192  }
193 
194  // Add the new object to the Object Pool
195  _poolID = _poolReference.Add (newWebObject);
196 
197  // We need to keep this GameObject around as it is referenced for spawning.
198  newWebObject.transform.parent = _poolReference.gameObject.transform;
199  newWebObject.gameObject.SetActive (false);
200  _initialized = true;
201  }
202  }
