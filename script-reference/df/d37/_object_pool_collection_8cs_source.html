157  public GameObject Spawn(Vector3 location, Quaternion rotation)
158  {
159  if ( _pooledObjects != null && _pooledObjects.Length > 0 )
160  {
161  GameObject spawnedObject = _pooledObjects[0];
162 
163  if ( spawnedObject == null ) { Debug.LogError(
164  "[h] A GameObject (" + prefab.name + ") has been destroyed," +
165  "but is still referenced by the Object Pool. If you must destroy a GameObject manually," +
166  "please make sure to remove it from the pool as well:" +
167  "hObjectPool.Instance.Remove(GameObject);");
168 
169  return null;
170  }
171 
172  Hydrogen.Array.RemoveAt<GameObject>(ref _pooledObjects, 0);
173 
174  // Move and rotate before we call on spawned
175  spawnedObject.transform.position = location;
176  spawnedObject.transform.rotation = rotation;
177 
178  if ( manageParticles )
179  {
180  if (hasLegacyParticleAnimator) spawnedObject.GetComponent<ParticleAnimator>().autodestruct = false;
181  if (hasLegacyParticleEmitter) spawnedObject.GetComponent<ParticleEmitter>().emit = true;
182  }
183 
184  if ( hasObjectPoolItem )
185  {
186  //spawnedObject.GetComponent<ObjectPoolItemBase>().poolID = _poolID;
187 
188  spawnedObject.GetComponent<ObjectPoolItemBase>().OnSpawned();
189  }
190  else if ( sendMessage )
191  {
192  spawnedObject.SendMessage("OnSpawned", SendMessageOptions.DontRequireReceiver);
193  }
194  else
195  {
196  spawnedObject.SetActive(true);
197  }
198 
199  if ( trackSpawnedObjects )
200  {
201  // No already spawned objects
202  if (_spawnedObjects == null ) _spawnedObjects = new GameObject[0];
203 
204  Hydrogen.Array.Add<GameObject>(ref _spawnedObjects, spawnedObject, false);
205  }
206 
207  return spawnedObject;
208  }
209  else if ( spawnMoreIfNeeded )
210  {
211  AddToPool();
212  return Spawn(location, rotation);
213  }
214  else
215  {
216  return null;
217  }
218  }

225  public void Despawn(GameObject gameObject, bool onSpawn)
226  {
227  if ( !manageParticles ) gameObject.transform.parent = _parentTransform;
228 
229  // Has our handler
230  if ( hasObjectPoolItem )
231  {
232  if ( manageParticles )
233  {
234  gameObject.GetComponent<ObjectPoolItemBase>().DespawnSafely();
235  return;
236  }
237  else
238  {
239  gameObject.GetComponent<ObjectPoolItemBase>().OnDespawned();
240  }
241 
242  }
243  // Use slow method
244  else if (sendMessage )
245  {
246  if ( manageParticles )
247  {
248  gameObject.SendMessage("WaitToDespawn", SendMessageOptions.DontRequireReceiver);
249  return;
250  }
251  else
252  {
253  gameObject.SendMessage("OnDespawned", SendMessageOptions.DontRequireReceiver);
254  }
255  }
256  // Fail safe
257  else
258  {
259  // This stops things from keeping their velocity from previous
260  if ( hasRigidbody )
261  {
262  gameObject.rigidbody.velocity = Vector3.zero;
263  gameObject.rigidbody.angularVelocity = Vector3.zero;
264  }
265 
266  gameObject.SetActive(false);
267 
268  }
269 
270  if ( !onSpawn )
271  {
272  if ( trackSpawnedObjects ) Hydrogen.Array.Remove<GameObject>(ref _spawnedObjects, gameObject);
273  Hydrogen.Array.Add<GameObject>(ref _pooledObjects, gameObject, false);
274  }
275  }
